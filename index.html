from itl_policy_builder import PolicyBuilder, Effect, field

policy = (
    PolicyBuilder("require-westeurope")
    .display_name("Require West Europe Location")
    .description("Ensures all resources are deployed in West Europe")
    .category("General")
    .mode("All")
    .with_rule(
        if_=field("location").not_equals("westeurope"),
        then=Effect.DENY,
        message="Resources must be deployed in West Europe",
    )
    .build()
)

# Serialise to ARM JSON
print(policy.to_arm_json())

from itl_policy_builder import field, all_of, any_of, not_, count

# Combine multiple conditions
condition = all_of(
    field("location").in_("westeurope", "northeurope"),
    field("tags.environment").exists(),
    not_(field("type").equals("ITL.Core/resourceGroups")),
)

# Count-based conditions
condition = count("properties.ipConfigurations").where(
    field("properties.publicIpAddress").exists()
).equals(0)   # No public IPs allowed

# Operator overloads
condition = (
    field("location").not_equals("westeurope") &
    field("tags.environment").exists()          # AND
)

condition = field("sku").equals("Basic") | field("sku").equals("Standard")  # OR
condition = ~field("type").equals("ITL.Core/resourceGroups")                # NOT
Python
from itl_policy_builder import PolicyBuilder, PolicyAssignmentBuilder, Effect, field

# Policy with parameters
policy = (
    PolicyBuilder("allowed-locations")
    .display_name("Allowed Locations")
    .parameter(
        "allowedLocations",
        type="Array",
        display_name="Allowed Locations",
        description="List of allowed Azure regions",
        default=["westeurope", "northeurope"],
    )
    .parameter(
        "effect",
        type="String",
        default="Deny",
        allowed_values=["Deny", "Audit", "Disabled"],
    )
    .with_rule(
        if_=field("location").not_in("[parameters('allowedLocations')]"),
        then=Effect.DENY,
    )
    .build()
)

# Policy assignment — override parameters at assignment time
assignment = (
    PolicyAssignmentBuilder("enforce-locations-prod")
    .policy_definition_id(policy.id)
    .scope("/subscriptions/sub-prod-001")
    .display_name("Enforce Locations in Production")
    .parameter("allowedLocations", ["westeurope"])         # narrow scope in prod
    .exclude_scope("/subscriptions/sub-prod-001/resourceGroups/legacy-rg")
    .non_compliance_message("This resource violates the location policy")
    .build()
)

from itl_policy_builder import PolicyEvaluator, PolicyBuilder, field, Effect

evaluator = PolicyEvaluator()

evaluator.register_assignment(assignment)

resource = {
    "id": "/subscriptions/sub-001/resourceGroups/rg-1/providers/ITL.Compute/virtualMachines/vm-1",

    "type": "ITL.Compute/virtualMachines",
    "location": "eastus",
}

if result.denied:

Generate & Deploy to Talos

from itl_policy_builder.templates.kyverno import get_talos_security_bundle
from itl_policy_builder.deploy import PolicyDeployer, DeployConfig, DeployTarget

# Get all Talos security policies
policies = get_talos_security_bundle()

# Deploy to the cluster
config = DeployConfig(
    target=DeployTarget.KUBERNETES,
    kube_context="talos-prod",
    action="audit",
)
deployer = PolicyDeployer(configs=[config])
results = asyncio.run(deployer.deploy(
    [p.to_yaml() for p in policies]
))

Manage the full policy lifecycle without writing Python. Install with the [cli] extra.

List available policy templates by category

generate

Generate policies in Kyverno or Azure ARM format

validate

Validate policy files before deployment

deploy

Deploy to Kubernetes, Azure, or ITL Control Plane

Initialise the CLI configuration file

explain

Explain templates or Azure governance concepts (--about)

inventory

Inventory assignments and definitions across subscriptions

describe

Fetch live Azure details for a specific governance resource

publish

Publish a policy bundle to ITL catalog, OCI registry, or Git repository

# List templates
itl-policy list --category talos

# Generate Kyverno policies (Kubernetes)
itl-policy generate --template talos-security --style kyverno --output k8s.yaml

# Generate Azure ARM policies
itl-policy generate --template talos-security --style azure --format json --output azure.json

# Generate PQC transition initiative
itl-policy generate --template pqc-transition --style azure --format json --output pqc.json
# Deploy to Kubernetes (audit mode)
itl-policy deploy --file k8s.yaml --target kubernetes --action audit

# Deploy to Azure (enforce mode)
itl-policy deploy --file azure.json --target azure \
  --subscription-id "00000000-0000-0000-0000-000000000000" \
  --assignment-scope "/subscriptions/00000000-0000-0000-0000-000000000000" \
  --action enforce

# Deploy with Azure CLI authentication
itl-policy deploy --file azure.json --target azure \
  --subscription-id "<sub-id>" \
  --assignment-scope "/subscriptions/<sub-id>" \
  --azure-auth cli
# Inventory all policies in a subscription
itl-policy inventory --subscription-id "<sub-id>"

# Inventory across all accessible subscriptions
itl-policy inventory --all-subscriptions --format table

# Export inventory to JSON
itl-policy inventory --subscription-id "<sub-id>" \
  --include assignments --include definitions \
  --format json --output inventory.json
# Explain what policies a template contains
itl-policy explain --template cis-azure --section AKS

# Explain Azure governance concepts
itl-policy explain --about management-group
itl-policy explain --about policy-definition
itl-policy explain --about policy-assignment

# Describe a live Azure resource
itl-policy describe subscription "Production" --format json
itl-policy describe policy-assignment cis-azure-baseline \
  --subscription-id "<sub-id>"
# Publish a bundle to the ITL Control Plane policy catalog
itl-policy publish -f azure.json --name cis-azure-baseline --version 2.0.0

# Dry-run to preview what would be published
itl-policy publish -f azure.json --name cis-azure-baseline --version 2.0.0 --dry-run

# Add extra tags to a bundle
itl-policy publish -f azure.json --name cis-azure-baseline --version 2.0.0 \
  --tag cis --tag azure --tag production

# Publish to an OCI artifact registry (e.g. GHCR)
itl-policy publish -f k8s.yaml --registry oci \
  --oci-registry ghcr.io/itlusions \
  --name kyverno-talos-security --version 1.2.0

# Publish to a Git repository (policies-as-code)
itl-policy publish -f azure.json --registry git \
  --git-repo https://github.com/org/policy-store.git \
  --git-branch main --git-path policies \
  --name cis-azure-baseline --version 2.0.0

Azure Deployment

Deploy ARM-compatible policies directly to Azure subscriptions and management groups using the native Azure SDK.

Python SDK

import asyncio
from itl_policy_builder.deploy import (
    PolicyDeployer, DeployConfig,
    DeployTarget, DeployAction,
)

config = DeployConfig(
    target=DeployTarget.AZURE,
    azure_subscription_id="00000000-...",
    azure_assignment_scope="/subscriptions/00000000-...",
    azure_credential="default",
    action=DeployAction.ENFORCE,
    dry_run=False,
)

deployer = PolicyDeployer(configs=[config])
results = asyncio.run(deployer.deploy([policy.to_arm_dict()]))
for r in results:
    print(r.summary)
